```python
# src/run_weekly_analysis.py

import os
from dotenv import load_dotenv
from datetime import datetime

from data_collector.pipiads_client import PiPiAdsClient
from database.storage import ProductDatabase
from ai_agent.trend_detector import TrendDetector
from ai_agent.likato_fit_analyzer import LikatoFitAnalyzer
from ai_agent.ssr_tester import SSRTester
from ai_agent.orchestrator import AIOrchestrator
from reports.generator import generate_weekly_report

load_dotenv()

def main():
    print(f"\n{'='*60}")
    print(f"üöÄ ProductRadar Weekly Analysis")
    print(f"–î–∞—Ç–∞: {datetime.now().strftime('%Y-%m-%d %H:%M')}")
    print(f"{'='*60}\n")
    
    # –ö–æ–Ω—Ñ–∏–≥—É—Ä–∞—Ü–∏—è
    REGIONS = ["US", "TH", "VN"]
    CATEGORY = "Beauty & Personal Care"
    
    # 1. –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è –∫–ª–∏–µ–Ω—Ç–æ–≤
    print("üì¶ –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è...")
    pipiads = PiPiAdsClient(os.getenv("PIPIADS_API_KEY"))
    db = ProductDatabase()
    
    orchestrator = AIOrchestrator(
        trend_detector=TrendDetector(db),
        likato_analyzer=LikatoFitAnalyzer(os.getenv("OPENAI_API_KEY")),
        ssr_tester=SSRTester(os.getenv("OPENAI_API_KEY"))
    )
    
    # 2. –°–±–æ—Ä –¥–∞–Ω–Ω—ã—Ö
    print(f"\nüì• –°–±–æ—Ä –¥–∞–Ω–Ω—ã—Ö –∏–∑ PiPiAds...")
    all_products = []
    
    for region in REGIONS:
        print(f"  - {region}...", end=" ")
        products = pipiads.get_trending_products(
            region=region,
            category=CATEGORY,
            time_range="7d",
            limit=50
        )
        all_products.extend(products)
        print(f"‚úì {len(products)} –ø—Ä–æ–¥—É–∫—Ç–æ–≤")
        
        # –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ –≤ –ë–î
        for product in products:
            db.save_product(product)
    
    print(f"\n  üíæ –í—Å–µ–≥–æ —Å–æ–±—Ä–∞–Ω–æ: {len(all_products)} –ø—Ä–æ–¥—É–∫—Ç–æ–≤")
    
    # 3. –ò–ò-–∞–Ω–∞–ª–∏–∑
    print(f"\nü§ñ –ò–ò-–∞–Ω–∞–ª–∏–∑...")
    results = orchestrator.analyze_weekly_opportunities(all_products)
    
    # 4. –ì–µ–Ω–µ—Ä–∞—Ü–∏—è –æ—Ç—á—ë—Ç–∞
    print(f"\nüìä –ì–µ–Ω–µ—Ä–∞—Ü–∏—è –æ—Ç—á—ë—Ç–∞...")
    report = generate_weekly_report(results)
    
    # –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ
    week_num = datetime.now().strftime('%Y_%W')
    report_path = f"reports/week_{week_num}.md"
    
    os.makedirs("reports", exist_ok=True)
    with open(report_path, "w", encoding="utf-8") as f:
        f.write(report)
    
    print(f"  ‚úì –û—Ç—á—ë—Ç —Å–æ—Ö—Ä–∞–Ω—ë–Ω: {report_path}")
    
    # –í—ã–≤–æ–¥ —Å–∞–º–º–∞—Ä–∏ –≤ –∫–æ–Ω—Å–æ–ª—å
    print(f"\n{'='*60}")
    print(results['summary'])
    print(f"{'='*60}\n")
    
    print("‚úÖ –ê–Ω–∞–ª–∏–∑ –∑–∞–≤–µ—Ä—à—ë–Ω!")

if __name__ == "__main__":
    main()
```
