**–ü—Ä–∏–º–µ—Ä –∑–∞–ø—Ä–æ—Å–∞:**

SELECT 
    p.id, p.name, p.category,
    start.views as start_views,
    end.views as end_views,
    (end.views - start.views) as growth
FROM products p
JOIN metrics_history start ON p.id = start.product_id 
    AND start.date = date('now', '-7 days')
JOIN metrics_history end ON p.id = end.product_id 
    AND end.date = date('now')
ORDER BY growth DESC
LIMIT 20;
