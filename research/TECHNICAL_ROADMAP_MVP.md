```python
# src/data_collector/pipiads_client.py

from dataclasses import dataclass
from datetime import datetime, timedelta
from typing import List, Optional

@dataclass
class Product:
    """–ú–æ–¥–µ–ª—å –ø—Ä–æ–¥—É–∫—Ç–∞"""
    id: str
    name: str
    price: float
    category: str
    region: str
    shop_url: str
    
    # –ú–µ—Ç—Ä–∏–∫–∏
    views: int
    sales: int
    revenue_estimate: float
    engagement_rate: float
    video_count: int
    
    # –¢—Ä–µ–Ω–¥
    trend_status: str  # rising | stable | declining
    first_seen: datetime
    
    # –î–æ–ø–æ–ª–Ω–∏—Ç–µ–ª—å–Ω–æ
    images: List[str]
    description: Optional[str] = None

class PiPiAdsClient:
    """–ö–ª–∏–µ–Ω—Ç –¥–ª—è —Ä–∞–±–æ—Ç—ã —Å PiPiAds API"""
    
    def __init__(self, api_key: str):
        self.api_key = api_key
        self.base_url = "https://api.pipiads.com/v1"
        self.headers = {
            "Authorization": f"Bearer {api_key}",
            "Content-Type": "application/json"
        }
    
    def get_trending_products(
        self, 
        region: str = "US",
        category: str = "Beauty & Personal Care",
        time_range: str = "7d",
        limit: int = 50
    ) -> List[Product]:
        """–ü–æ–ª—É—á–∏—Ç—å —Ç—Ä–µ–Ω–¥–æ–≤—ã–µ –ø—Ä–æ–¥—É–∫—Ç—ã"""
        
        response = requests.get(
            f"{self.base_url}/tiktok-shop/products/trending",
            headers=self.headers,
            params={
                "region": region,
                "category": category,
                "sort_by": "views",
                "time_range": time_range,
                "limit": limit
            }
        )
        
        data = response.json()
        return [self._parse_product(p) for p in data["products"]]
    
    def get_product_details(self, product_id: str) -> dict:
        """–ü–æ–ª—É—á–∏—Ç—å –¥–µ—Ç–∞–ª—å–Ω—É—é –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏—é –æ –ø—Ä–æ–¥—É–∫—Ç–µ"""
        
        response = requests.get(
            f"{self.base_url}/tiktok-shop/product/{product_id}",
            headers=self.headers,
            params={
                "include_videos": True,
                "include_history": True
            }
        )
        
        return response.json()
    
    def search_products(self, query: str, region: str = "US") -> List[Product]:
        """–ü–æ–∏—Å–∫ –ø—Ä–æ–¥—É–∫—Ç–æ–≤ –ø–æ –∫–ª—é—á–µ–≤—ã–º —Å–ª–æ–≤–∞–º"""
        
        response = requests.get(
            f"{self.base_url}/tiktok-shop/search",
            headers=self.headers,
            params={
                "query": query,
                "region": region,
                "category": "Beauty & Personal Care"
            }
        )
        
        data = response.json()
        return [self._parse_product(p) for p in data["products"]]
    
    def _parse_product(self, raw: dict) -> Product:
        """–ü–∞—Ä—Å–∏–Ω–≥ —Å—ã—Ä—ã—Ö –¥–∞–Ω–Ω—ã—Ö –≤ –º–æ–¥–µ–ª—å Product"""
        return Product(
            id=raw["id"],
            name=raw["name"],
            price=raw["price"],
            category=raw["category"],
            region=raw.get("region", "US"),
            shop_url=raw["shop_url"],
            views=raw["metrics"]["views"],
            sales=raw["metrics"]["sales"],
            revenue_estimate=raw["metrics"]["revenue_estimate"],
            engagement_rate=raw["metrics"]["engagement_rate"],
            video_count=raw["metrics"]["video_count"],
            trend_status=raw["trend_status"],
            first_seen=datetime.fromisoformat(raw["first_seen"]),
            images=raw["images"],
            description=raw.get("description")
        )
```

```python
# src/database/storage.py

import sqlite3
from datetime import datetime
from typing import List
import json

class ProductDatabase:
    """SQLite –±–∞–∑–∞ –¥–ª—è —Ö—Ä–∞–Ω–µ–Ω–∏—è –ø—Ä–æ–¥—É–∫—Ç–æ–≤ –∏ –∏—Å—Ç–æ—Ä–∏–∏"""
    
    def __init__(self, db_path: str = "data/products.db"):
        self.db_path = db_path
        self._init_db()
    
    def _init_db(self):
        """–°–æ–∑–¥–∞–Ω–∏–µ —Ç–∞–±–ª–∏—Ü"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        # –¢–∞–±–ª–∏—Ü–∞ –ø—Ä–æ–¥—É–∫—Ç–æ–≤
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS products (
                id TEXT PRIMARY KEY,
                name TEXT,
                price REAL,
                category TEXT,
                region TEXT,
                shop_url TEXT,
                first_seen TEXT,
                last_updated TEXT,
                images TEXT,
                description TEXT
            )
        ''')
        
        # –¢–∞–±–ª–∏—Ü–∞ –º–µ—Ç—Ä–∏–∫ (–∏—Å—Ç–æ—Ä–∏—è)
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS metrics_history (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                product_id TEXT,
                date TEXT,
                views INTEGER,
                sales INTEGER,
                revenue_estimate REAL,
                engagement_rate REAL,
                video_count INTEGER,
                trend_status TEXT,
                FOREIGN KEY (product_id) REFERENCES products(id)
            )
        ''')
        
        # –ò–Ω–¥–µ–∫—Å—ã –¥–ª—è –±—ã—Å—Ç—Ä–æ–≥–æ –ø–æ–∏—Å–∫–∞
        cursor.execute('''
            CREATE INDEX IF NOT EXISTS idx_product_region 
            ON products(region)
        ''')
        
        cursor.execute('''
            CREATE INDEX IF NOT EXISTS idx_metrics_date 
            ON metrics_history(date)
        ''')
        
        conn.commit()
        conn.close()
    
    def save_product(self, product: Product):
        """–°–æ—Ö—Ä–∞–Ω–∏—Ç—å –ø—Ä–æ–¥—É–∫—Ç –∏ –µ–≥–æ –º–µ—Ç—Ä–∏–∫–∏"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        # Upsert –ø—Ä–æ–¥—É–∫—Ç–∞
        cursor.execute('''
            INSERT OR REPLACE INTO products 
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
        ''', (
            product.id,
            product.name,
            product.price,
            product.category,
            product.region,
            product.shop_url,
            product.first_seen.isoformat(),
            datetime.now().isoformat(),
            json.dumps(product.images),
            product.description
        ))
        
        # –î–æ–±–∞–≤–∏—Ç—å –º–µ—Ç—Ä–∏–∫–∏
        cursor.execute('''
            INSERT INTO metrics_history 
            (product_id, date, views, sales, revenue_estimate, 
             engagement_rate, video_count, trend_status)
            VALUES (?, ?, ?, ?, ?, ?, ?, ?)
        ''', (
            product.id,
            datetime.now().date().isoformat(),
            product.views,
            product.sales,
            product.revenue_estimate,
            product.engagement_rate,
            product.video_count,
            product.trend_status
        ))
        
        conn.commit()
        conn.close()
    
    def get_trending_growth(self, days: int = 7) -> List[dict]:
        """–ü–æ–ª—É—á–∏—Ç—å –ø—Ä–æ–¥—É–∫—Ç—ã —Å –Ω–∞–∏–±–æ–ª—å—à–∏–º —Ä–æ—Å—Ç–æ–º –∑–∞ –ø–µ—Ä–∏–æ–¥"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        # –°—Ä–∞–≤–Ω–∏–≤–∞–µ–º –º–µ—Ç—Ä–∏–∫–∏ –Ω–∞—á–∞–ª–∞ –∏ –∫–æ–Ω—Ü–∞ –ø–µ—Ä–∏–æ–¥–∞
        query = f'''
            WITH start_metrics AS (
                SELECT product_id, views as start_views, sales as start_sales
                FROM metrics_history
                WHERE date = date('now', '-{days} days')
            ),
            end_metrics AS (
                SELECT product_id, views as end_views, sales as end_sales
                FROM metrics_history
                WHERE date = date('now')
            )
            SELECT 
                p.id, p.name, p.price, p.category, p.region,
                s.start_views, e.end_views, 
                (e.end_views - s.start_views) as views_growth,
                s.start_sales, e.end_sales,
                (e.end_sales - s.start_sales) as sales_growth
            FROM products p
            JOIN start_metrics s ON p.id = s.product_id
            JOIN end_metrics e ON p.id = e.product_id
            ORDER BY views_growth DESC
        '''
        
        cursor.execute(query)
        results = cursor.fetchall()
        conn.close()
        
        return results
```

### Module 1: Trend Detection (–î–µ—Ç–µ–∫—Ç–æ—Ä —Ç—Ä–µ–Ω–¥–æ–≤)

```python
# src/ai_agent/trend_detector.py

from typing import List, Dict
import numpy as np
from sklearn.linear_model import LinearRegression

class TrendDetector:
    """–ê–Ω–∞–ª–∏–∑–∏—Ä—É–µ—Ç —Ç—Ä–µ–Ω–¥—ã –∏ –æ–ø—Ä–µ–¥–µ–ª—è–µ—Ç —Å—Ç–∞–¥–∏—é –∂–∏–∑–Ω–µ–Ω–Ω–æ–≥–æ —Ü–∏–∫–ª–∞"""
    
    def analyze_trends(self, products: List[Product]) -> List[Dict]:
        """
        –ö–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏—è –ø—Ä–æ–¥—É–∫—Ç–æ–≤ –ø–æ —Å—Ç–∞–¥–∏—è–º —Ç—Ä–µ–Ω–¥–∞:
        - üöÄ Early (—Ä–∞–Ω–Ω—è—è —Å—Ç–∞–¥–∏—è) - —Ç–æ–ª—å–∫–æ –Ω–∞—á–∏–Ω–∞–µ—Ç –≤–∑–ª–µ—Ç–∞—Ç—å
        - üìà Rising (—Ä–∞—Å—Ç—É—â–∏–π) - –∞–∫—Ç–∏–≤–Ω—ã–π —Ä–æ—Å—Ç
        - üî• Peak (–ø–∏–∫) - –º–∞–∫—Å–∏–º–∞–ª—å–Ω–∞—è –ø–æ–ø—É–ª—è—Ä–Ω–æ—Å—Ç—å
        - üìâ Declining (—Å–ø–∞–¥) - —É–∂–µ –ø—Ä–æ—Ö–æ–¥–∏—Ç
        """
        
        results = []
        
        for product in products:
            # –ü–æ–ª—É—á–∞–µ–º –∏—Å—Ç–æ—Ä–∏—é –º–µ—Ç—Ä–∏–∫
            history = self.db.get_metrics_history(product.id, days=14)
            
            if len(history) < 3:
                continue  # –Ω–µ–¥–æ—Å—Ç–∞—Ç–æ—á–Ω–æ –¥–∞–Ω–Ω—ã—Ö
            
            # –ê–Ω–∞–ª–∏–∑ –¥–∏–Ω–∞–º–∏–∫–∏
            trend_data = self._analyze_growth_pattern(history)
            
            # –ö–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏—è —Å—Ç–∞–¥–∏–∏
            stage = self._classify_stage(trend_data)
            
            # –ü—Ä–æ–≥–Ω–æ–∑ –Ω–∞ —Å–ª–µ–¥—É—é—â—É—é –Ω–µ–¥–µ–ª—é
            forecast = self._forecast_next_week(history)
            
            results.append({
                "product": product,
                "stage": stage,
                "growth_rate": trend_data["growth_rate"],
                "acceleration": trend_data["acceleration"],
                "forecast_views": forecast["views"],
                "forecast_sales": forecast["sales"],
                "opportunity_score": self._calculate_opportunity_score(
                    stage, trend_data, forecast
                )
            })
        
        return results
    
    def _analyze_growth_pattern(self, history: List[dict]) -> dict:
        """–ê–Ω–∞–ª–∏–∑ –ø–∞—Ç—Ç–µ—Ä–Ω–∞ —Ä–æ—Å—Ç–∞"""
        
        dates = np.array([h["date"] for h in history])
        views = np.array([h["views"] for h in history])
        
        # –°–∫–æ—Ä–æ—Å—Ç—å —Ä–æ—Å—Ç–∞ (–ø–µ—Ä–≤–∞—è –ø—Ä–æ–∏–∑–≤–æ–¥–Ω–∞—è)
        growth_rate = np.gradient(views)
        
        # –£—Å–∫–æ—Ä–µ–Ω–∏–µ (–≤—Ç–æ—Ä–∞—è –ø—Ä–æ–∏–∑–≤–æ–¥–Ω–∞—è)
        acceleration = np.gradient(growth_rate)
        
        # –°—Ä–µ–¥–Ω—è—è —Å–∫–æ—Ä–æ—Å—Ç—å —Ä–æ—Å—Ç–∞ –∑–∞ –ø–æ—Å–ª–µ–¥–Ω–∏–µ 7 –¥–Ω–µ–π
        avg_growth = np.mean(growth_rate[-7:])
        
        # –ö–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç –≤–∞—Ä–∏–∞—Ü–∏–∏ (—Å—Ç–∞–±–∏–ª—å–Ω–æ—Å—Ç—å —Ä–æ—Å—Ç–∞)
        cv = np.std(growth_rate) / np.mean(growth_rate) if np.mean(growth_rate) > 0 else 0
        
        return {
            "growth_rate": avg_growth,
            "acceleration": np.mean(acceleration[-3:]),
            "stability": 1 / (1 + cv),  # 0-1, —á–µ–º –≤—ã—à–µ - —Ç–µ–º —Å—Ç–∞–±–∏–ª—å–Ω–µ–µ
            "total_growth": (views[-1] - views[0]) / views[0] if views[0] > 0 else 0
        }
    
    def _classify_stage(self, trend_data: dict) -> str:
        """–ö–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏—è —Å—Ç–∞–¥–∏–∏ —Ç—Ä–µ–Ω–¥–∞"""
        
        growth_rate = trend_data["growth_rate"]
        acceleration = trend_data["acceleration"]
        
        if growth_rate > 100000 and acceleration > 0:
            return "early"  # üöÄ –†–∞–Ω–Ω—è—è —Å—Ç–∞–¥–∏—è - –±—ã—Å—Ç—Ä—ã–π —Ä–æ—Å—Ç —Å —É—Å–∫–æ—Ä–µ–Ω–∏–µ–º
        elif growth_rate > 50000 and acceleration >= 0:
            return "rising"  # üìà –†–∞—Å—Ç—É—â–∏–π - —Å—Ç–∞–±–∏–ª—å–Ω—ã–π —Ä–æ—Å—Ç
        elif growth_rate > 0 and acceleration < 0:
            return "peak"  # üî• –ü–∏–∫ - —Ä–æ—Å—Ç –∑–∞–º–µ–¥–ª—è–µ—Ç—Å—è
        else:
            return "declining"  # üìâ –°–ø–∞–¥
    
    def _forecast_next_week(self, history: List[dict]) -> dict:
        """–ü—Ä–æ—Å—Ç–æ–π –ø—Ä–æ–≥–Ω–æ–∑ –Ω–∞ —Å–ª–µ–¥—É—é—â—É—é –Ω–µ–¥–µ–ª—é"""
        
        if len(history) < 7:
            return {"views": history[-1]["views"], "sales": history[-1]["sales"]}
        
        # –õ–∏–Ω–µ–π–Ω–∞—è —Ä–µ–≥—Ä–µ—Å—Å–∏—è –¥–ª—è –ø—Ä–æ–≥–Ω–æ–∑–∞
        X = np.array(range(len(history))).reshape(-1, 1)
        y_views = np.array([h["views"] for h in history])
        y_sales = np.array([h["sales"] for h in history])
        
        model_views = LinearRegression().fit(X, y_views)
        model_sales = LinearRegression().fit(X, y_sales)
        
        next_week = len(history) + 7
        forecast_views = model_views.predict([[next_week]])[0]
        forecast_sales = model_sales.predict([[next_week]])[0]
        
        return {
            "views": max(0, int(forecast_views)),
            "sales": max(0, int(forecast_sales))
        }
    
    def _calculate_opportunity_score(
        self, stage: str, trend_data: dict, forecast: dict
    ) -> float:
        """
        –û—Ü–µ–Ω–∫–∞ –ø–µ—Ä—Å–ø–µ–∫—Ç–∏–≤–Ω–æ—Å—Ç–∏ –¥–ª—è Likato (0-100)
        
        –í—ã—Å–æ–∫–∏–π —Å–∫–æ—Ä = –≤—ã–≥–æ–¥–Ω–æ –ø—Ä–æ–∏–∑–≤–æ–¥–∏—Ç—å –∞–Ω–∞–ª–æ–≥
        """
        
        score = 0
        
        # –ë–æ–Ω—É—Å –∑–∞ —Å—Ç–∞–¥–∏—é
        stage_bonus = {
            "early": 40,    # –õ—É—á—à–µ–µ –≤—Ä–µ–º—è –¥–ª—è –≤—Ö–æ–¥–∞
            "rising": 30,   # –•–æ—Ä–æ—à–µ–µ –≤—Ä–µ–º—è
            "peak": 10,     # –£–∂–µ –ø–æ–∑–¥–Ω–æ–≤–∞—Ç–æ
            "declining": 0  # –ù–µ —Å—Ç–æ–∏—Ç
        }
        score += stage_bonus[stage]
        
        # –ë–æ–Ω—É—Å –∑–∞ —Å—Ç–∞–±–∏–ª—å–Ω–æ—Å—Ç—å —Ä–æ—Å—Ç–∞
        score += trend_data["stability"] * 20
        
        # –ë–æ–Ω—É—Å –∑–∞ –ø—Ä–æ–≥–Ω–æ–∑–∏—Ä—É–µ–º—ã–π –æ–±—ä—ë–º –ø—Ä–æ–¥–∞–∂
        if forecast["sales"] > 10000:
            score += 20
        elif forecast["sales"] > 5000:
            score += 10
        
        # –ë–æ–Ω—É—Å –∑–∞ —Å–∫–æ—Ä–æ—Å—Ç—å —Ä–æ—Å—Ç–∞
        if trend_data["growth_rate"] > 100000:
            score += 20
        elif trend_data["growth_rate"] > 50000:
            score += 10
        
        return min(100, score)
```

```python
            # –§–∏–ª—å—Ç—Ä: —Ç–æ–ª—å–∫–æ early/rising —Å—Ç–∞–¥–∏–∏
            if trend_analysis["stage"] in ["declining", "peak"]:
                continue
            
            # 2. –ê–Ω–∞–ª–∏–∑ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–∏—è Likato
            fit_analysis = self.likato_analyzer.analyze_fit(
                product, trend_analysis
            )
            
            # –§–∏–ª—å—Ç—Ä: fit_score >= 60
            if fit_analysis["fit_score"] < 60:
                continue
            
            # 3. SSR —Ç–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ (–æ–ø—Ü–∏–æ–Ω–∞–ª—å–Ω–æ, –¥–ª—è —Ç–æ–ø –∫–∞–Ω–¥–∏–¥–∞—Ç–æ–≤)
            # –ü—Ä–æ–ø—É—Å–∫–∞–µ–º –¥–ª—è MVP, —á—Ç–æ–±—ã –Ω–µ —Ç—Ä–∞—Ç–∏—Ç—å –º–Ω–æ–≥–æ API –∑–∞–ø—Ä–æ—Å–æ–≤
            # –ú–æ–∂–Ω–æ –≤–∫–ª—é—á–∏—Ç—å –¥–ª—è —Ñ–∏–Ω–∞–ª—å–Ω–æ–≥–æ —Ç–æ–ø-10
            
            # –í—ã—á–∏—Å–ª—è–µ–º –æ–±—â–∏–π —Å–∫–æ—Ä
            overall_score = self._calculate_overall_score(
                trend_analysis, fit_analysis
            )
            
            results.append({
                "product": product,
                "overall_score": overall_score,
                "trend_analysis": trend_analysis,
                "fit_analysis": fit_analysis,
                "ssr_results": None  # –ø–æ–∫–∞ –æ—Ç–∫–ª—é—á–µ–Ω–æ
            })
        
        # –°–æ—Ä—Ç–∏—Ä—É–µ–º –ø–æ –æ–±—â–µ–º—É —Å–∫–æ—Ä—É
        results.sort(key=lambda x: x["overall_score"], reverse=True)
        
        # –¢–æ–ø-10
        top_opportunities = results[:10]
        
        # –î–ª—è —Ç–æ–ø-3 –¥–µ–ª–∞–µ–º SSR —Ç–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ
        print("\nüß™ SSR —Ç–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ —Ç–æ–ø-3 –∫–∞–Ω–¥–∏–¥–∞—Ç–æ–≤...")
        for i, opp in enumerate(top_opportunities[:3]):
            product = opp["product"]
            
            # –§–æ—Ä–º–∏—Ä—É–µ–º –∫–æ–Ω—Ü–µ–ø—Ç –ø—Ä–æ–¥—É–∫—Ç–∞ –¥–ª—è Likato
            concept = self._create_likato_concept(
                product, opp["fit_analysis"]["adaptations"]
            )
            
            # –¢–µ—Å—Ç–∏—Ä—É–µ–º
            ssr_results = self.ssr_tester.test_concept(
                concept["name"],
                concept["description"],
                concept["price"]
            )
            
            opp["ssr_results"] = ssr_results
            opp["overall_score"] += ssr_results["average_rating"] * 5  # –±–æ–Ω—É—Å –∑–∞ SSR
        
        # –ü–µ—Ä–µ—Å–æ—Ä—Ç–∏—Ä—É–µ–º —Å —É—á—ë—Ç–æ–º SSR
        top_opportunities.sort(key=lambda x: x["overall_score"], reverse=True)
        
        return {
            "total_analyzed": len(products),
            "qualified_opportunities": len(results),
            "top_10": top_opportunities,
            "summary": self._generate_summary(top_opportunities)
        }
    
    def _calculate_overall_score(
        self, trend_analysis: dict, fit_analysis: dict
    ) -> float:
        """
        –í—ã—á–∏—Å–ª–µ–Ω–∏–µ –æ–±—â–µ–≥–æ —Å–∫–æ—Ä–∞ –≤–æ–∑–º–æ–∂–Ω–æ—Å—Ç–∏
        
        –§–æ—Ä–º—É–ª–∞:
        overall_score = opportunity_score * 0.4 + fit_score * 0.6
        
        fit_score –≤–∞–∂–Ω–µ–µ, —Ç.–∫. —Ç—Ä–µ–Ω–¥ –±–µ–∑ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–∏—è –±—Ä–µ–Ω–¥—É –±–µ—Å–ø–æ–ª–µ–∑–µ–Ω
        """
        return (
            trend_analysis["opportunity_score"] * 0.4 +
            fit_analysis["fit_score"] * 0.6
        )
    
    def _create_likato_concept(
        self, product: Product, adaptations: List[str]
    ) -> dict:
        """–°–æ–∑–¥–∞—Ç—å –∫–æ–Ω—Ü–µ–ø—Ç –ø—Ä–æ–¥—É–∫—Ç–∞ –¥–ª—è Likato –Ω–∞ –æ—Å–Ω–æ–≤–µ —Ç—Ä–µ–Ω–¥–∞"""
        
        # –ë–µ—Ä—ë–º –æ—Ä–∏–≥–∏–Ω–∞–ª—å–Ω—ã–π –ø—Ä–æ–¥—É–∫—Ç –∏ –∞–¥–∞–ø—Ç–∏—Ä—É–µ–º –ø–æ–¥ Likato
        concept_name = f"Likato {product.category} - –≤–¥–æ—Ö–Ω–æ–≤–ª—ë–Ω {product.name}"
        
        concept_description = f"""
–ù–∞—Ç—É—Ä–∞–ª—å–Ω—ã–π {product.category.lower()} –Ω–∞ –æ—Å–Ω–æ–≤–µ —Ç—Ä–µ–Ω–¥–æ–≤–æ–≥–æ –ø—Ä–æ–¥—É–∫—Ç–∞.
–ê–¥–∞–ø—Ç–∞—Ü–∏–∏ –¥–ª—è Likato:
{chr(10).join(f'- {a}' for a in adaptations)}
"""
        
        # –¶–µ–Ω–∞ –≤ —Ä—É–±–ª—è—Ö (–ø—Ä–∏–º–µ—Ä–Ω–æ)
        concept_price = product.price * 90 * 1.2  # USD to RUB + –Ω–∞—Ü–µ–Ω–∫–∞
        
        return {
            "name": concept_name,
            "description": concept_description,
            "price": concept_price
        }
    
    def _generate_summary(self, top_opportunities: List[dict]) -> str:
        """–ì–µ–Ω–µ—Ä–∞—Ü–∏—è —Å–∞–º–º–∞—Ä–∏ –¥–ª—è –æ—Ç—á—ë—Ç–∞"""
        
        summary = f"""
üìä –ï–ñ–ï–ù–ï–î–ï–õ–¨–ù–´–ô –û–¢–ß–Å–¢ PRODUCTRADA

–ù–∞–π–¥–µ–Ω–æ —Ç–æ–ø-{len(top_opportunities)} –≤–æ–∑–º–æ–∂–Ω–æ—Å—Ç–µ–π –¥–ª—è Likato.

ü•á –¢–û–ü-3 –ü–†–ò–û–†–ò–¢–ï–¢–ù–´–•:

"""
        for i, opp in enumerate(top_opportunities[:3], 1):
            product = opp["product"]
            fit = opp["fit_analysis"]
            trend = opp["trend_analysis"]
            
            summary += f"""
{i}. {product.name}
   –û–±—â–∏–π —Å–∫–æ—Ä: {opp['overall_score']:.1f}/100
   –°—Ç–∞–¥–∏—è —Ç—Ä–µ–Ω–¥–∞: {trend['stage']} ({trend['opportunity_score']:.0f}/100)
   –°–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–∏–µ Likato: {fit['fit_score']}/100
   –†–µ–∫–æ–º–µ–Ω–¥–∞—Ü–∏—è: {fit['recommendation']}
"""
            
            if opp["ssr_results"]:
                ssr = opp["ssr_results"]
                summary += f"""   SSR —Ç–µ—Å—Ç: {ssr['average_rating']:.1f}/5 ‚≠ê, {ssr['purchase_intent_pct']:.0f}% –Ω–∞–º–µ—Ä–µ–Ω—ã –∫—É–ø–∏—Ç—å

        return summary
```
